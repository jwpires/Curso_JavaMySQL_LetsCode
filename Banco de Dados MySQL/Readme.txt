CREATE TABLE CLIENTE (
COD_CLI INT AUTO_INCREMENT PRIMARY KEY,
NOME_CLI VARCHAR(40) NOT NULL,
ENDERECO VARCHAR(80) NOT NULL,
CIDADE VARCHAR (40) NOT NULL,
CEP VARCHAR(20) NOT NULL,
UF CHAR(2) NOT NULL
);

CREATE TABLE VENDEDOR (
COD_VEND INT AUTO_INCREMENT PRIMARY KEY,
NOME_VEND VARCHAR(40) NOT NULL,
SAL_FIXO REAL(7,2) NOT NULL,
FAIXA_COMISS REAL(5,2)
);

CREATE TABLE PEDIDO  (
cod_ped INT AUTO_INCREMENT PRIMARY KEY,
num_ped INT NOT NULL, 
prazo_entr TIME NOT NULL, 
cod_cli INT NOT NULL, 
cod_vend INT NOT NULL,
FOREIGN KEY (COD_CLI) REFERENCES CLIENTE (COD_CLI),
FOREIGN KEY (COD_VEND) REFERENCES VENDEDOR (COD_VEND)
);


CREATE TABLE PRODUTO (
cod_prod INT PRIMARY KEY auto_increment, 
unid_prod INT NOT NULL, 
desc_prod VARCHAR(40) NOT NULL, 
val_unit numeric(5,2) NOT NULL
);

CREATE TABLE ITEM_PEDIDO (
cod_item_ped INT PRIMARY KEY auto_increment,
cod_ped INT NOT NULL, 
cod_prod INT NOT NULL, 
qtd_ped INT NOT NULL,
CONSTRAINT Fk_COD_PEDIDO FOREIGN KEY (COD_PED) REFERENCES PEDIDO (COD_PED),
CONSTRAINT Fk_COD_PROD FOREIGN KEY (COD_PROD) REFERENCES PRODUTO (COD_PROD)
);
